## 6) TEST BLUEPRINTS (you must implement fully)

**RTL — mode propagation & value change**

```ts
it('ECOS↔NFL toggle updates all sections', async () => {
  render(<Dashboard />);
  const read = (id:string)=>screen.getByTestId(id).textContent?.trim()||'';
  const snap = () => ({
    tmv: read('metric-total-market-value'),
    ac:  read('metric-active-contracts'),
    avg: read('metric-avg-brand-value'),
    act: read('metric-market-activity'),
    tp:  read('mi-top-performers'),
    af:  read('mi-activity-feed'),
    qs:  read('quick-stats'),
    ss:  read('system-status'),
  });

  const ecos = snap();
  await userEvent.click(screen.getByTestId('toggle-nfl'));
  const nfl = snap();

  // Values must change in at least Financial Overview OR lists (and not be empty)
  expect(nfl.tmv && ecos.tmv && nfl.tmv !== ecos.tmv).toBe(true);
  expect(nfl.tp && ecos.tp && nfl.tp !== ecos.tp).toBe(true);

  await userEvent.click(screen.getByTestId('toggle-ecos'));
  const ecos2 = snap();
  expect(ecos2.tmv).toEqual(ecos.tmv);
});
```
